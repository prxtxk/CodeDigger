void solve() {
    int n; cin >> n;
    vector<int> a(n), b(n);

    for(int i=0; i<n; i++) cin >> a[i];
    for(int i=0; i<n; i++) cin >> b[i];

    vector<int> pre(n, 0);
    pre[0] = b[0];
    for(int i=1; i<n; i++){
        pre[i] = pre[i-1] + b[i];
    }

    int q; cin >> q;
    // cout << a[3] << " " << a[4] << " " << pre[4] << pre[3] <<  endl;
    while(q--){
        int l, r;
        cin >> l >> r;
        l--; r--;

        // cout << l << " " << r << endl;

        int ans = 0;
        for(int i=0; i<n-1; i++){
            if(i >= l && i <= r) continue;
            for(int j=i+1; j<n; j++){
                if(j >= l && j <= r) continue;
                int ans2 = 0;
                if((i < l && j < l) || (i > r && j > r)){
                    ans2 = pre[j] + a[i] + a[j];
                    ans2 -= pre[i];
                } else {
                    ans2 = pre[i] + pre[n-1] - pre[j] + a[i] + a[j];
                }
                ans = max(ans, ans2);
                // cout << i << " " << j << " " << ans << endl;
            }
        }

        cout << ans << endl;
    }
}

signed main() {
    fastio();
    int t = 1;
    // cin >> t;
    while (t--) {
        solve();
    }
    return 0;
}
